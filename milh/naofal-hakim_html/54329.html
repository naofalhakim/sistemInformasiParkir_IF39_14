6     package com.mycompany.impal;
7     
8     import com.mycompany.impal.view.LoginGUI;
9     import java.awt.event.ActionEvent;
10     import java.awt.event.ActionListener;
11     import javax.swing.JOptionPane;
12     
13     /**
14      *
15      * @author naofal
16      */
17     public class ControllerLogin implements ActionListener {
18     
19         private ApplicationModel model;
20         private LoginGUI view;
21     
22         public ControllerLogin() {
23             model = new ApplicationModel();
24             view = new LoginGUI();
25         }
26         
27         public void loadView(){
28             view.setVisible(true);
29             view.setActionListener(this);
30         }
31     
32         @Override
33         public void actionPerformed(ActionEvent e) {
34             Object source = e.getSource();
35     
36             String email, pass;
37             email = view.getJusername().getText();
38             pass = view.getJpassword().getText();
39     
40             if (source == view.getBtnDaftar()) {
41                 ControllerRegister controllerRegister = new ControllerRegister();
42                 view.setVisible(false);
43             } else if (source == view.getBtnLogin()) {
44     
45                 if (email == null || pass == null) {
46                     JOptionPane.showMessageDialog(view, "email atau password kosong");
47                 } else {
48                     String prio = model.cekLogin(email, pass);
49     
50                     if ("user".equals(prio)) {
51                         JOptionPane.showMessageDialog(view, "Login Berhasil");
52                         ApplicationModel.setIdPengendara(email);
53                         
54                         ControllerMenuUtama controllerMenuUtama = new ControllerMenuUtama();
55                         view.setVisible(false);
56                         
57                     } else if ("admin".equals(prio)) {
58                         JOptionPane.showMessageDialog(view, "Login Berhasil");
59                         
60                         ControllerAdmin controllerAdmin = new ControllerAdmin();
61                         view.setVisible(false);
62                     } else {
63                         JOptionPane.showMessageDialog(view, "Email Atau Password Salah");
64                     }
65     
66                 }
67             }
68         }
69     
70     }
